101 void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
102  auto x = H[i];
103  while (true) {
104  unsigned k = leftChild(i);
105  if (k >= H.size())
106  break;
107  if (k + 1 < H.size() && *H[k + 1] < *H[k])
108  ++k; // right child of i
109  if (!(*H[k] < *x))
110  break;
111  set(i, H[k]);
112  i = k;
113  }
114  set(i, x);
